### **2. Database Schema (Supabase)**

Before coding, you need a schema to track the "state" of your video generation. Run this in your Supabase SQL Editor:

```sql
-- Store the project/video request
create table videos (
  id uuid primary key default uuid_generate_v4(),
  user_id uuid references auth.users not null,
  prompt text not null,
  status text default 'planning', -- planning, waiting_approval, generating, rendering, completed, failed
  syllabus_doc_path text, -- Path to uploaded PDF in Storage
  final_video_url text,
  created_at timestamp with time zone default now()
);

-- Store individual scenes (generated by Planner/Script Agent)
create table scenes (
  id uuid primary key default uuid_generate_v4(),
  video_id uuid references videos not null,
  scene_order int not null,
  narration_script text,
  visual_plan text,
  manim_code text,
  is_rendered boolean default false,
  error_log text, -- To feed back to CodeGen Agent
  video_segment_url text
);

-- Enable Vector Extension for RAG
create extension if not exists vector;

```

### **4. The "Agentic Workflow" (LangGraph Logic)**

Here is the pseudo-code logic for the `workflow.py` file that Members 2 and 3 will build together. This matches your State Diagram .

```python
from langgraph.graph import StateGraph, END

# Define Nodes
workflow = StateGraph(AgentState)

workflow.add_node("retrieve_info", retrieve_rag_context) # Member 1
workflow.add_node("planner", generate_plan)              # Member 2
workflow.add_node("scripter", generate_scripts)          # Member 2
workflow.add_node("human_review", wait_for_approval)     # Member 2 (Breakpoint)
workflow.add_node("coder", generate_manim_code)          # Member 3
workflow.add_node("renderer", execute_in_sandbox)        # Member 4

# Define Edges (The Logic)
workflow.set_entry_point("retrieve_info")
workflow.add_edge("retrieve_info", "planner")
workflow.add_edge("planner", "scripter")
workflow.add_edge("scripter", "human_review")

# Conditional Edge after Render
def check_render_status(state):
    if state['last_render_error']:
        return "coder" # Loop back to fix code
    elif state['all_scenes_done']:
        return END
    else:
        return "coder" # Proceed to next scene

workflow.add_edge("human_review", "coder")
workflow.add_edge("coder", "renderer")
workflow.add_conditional_edges(
    "renderer",
    check_render_status
)

app = workflow.compile(interrupt_before=["human_review"])

```
